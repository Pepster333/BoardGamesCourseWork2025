Function Delivery(bool Pay, int GameAvailable)
	IF Pay is TRUE AND GameAvailable > 0
	THEN send email "order is being delievered. Thank you for choosing us!"
	AND set GameAvailable == GameAvailable -1
	Return GameAvailable
	END IF
